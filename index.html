async function getLeaderboard() { try { // 移除 created_at 的時間過濾，直接抓歷史最高前 10 名 const response = await fetch(`${SUB_URL}/rest/v1/scores?select=username,score&order=score.desc&limit=10`, { headers: { "apikey": SUB_KEY, "Authorization": `Bearer ${SUB_KEY}` } }); const data = await response.json(); const listEl = document.getElementById('leaderboard-list'); listEl.innerHTML = ''; if (!data || data.length === 0) { listEl.innerHTML = '

資料庫目前是空的，請確認上傳是否成功。

'; } else { data.forEach((item, index) => { const div = document.createElement('div'); div.className = 'leaderboard-item'; div.innerHTML = `第 ${index + 1} 名: ${item.username}${item.score} 分`; listEl.appendChild(div); }); } } catch (e) { console.error("載入榜單出錯:", e); } }
